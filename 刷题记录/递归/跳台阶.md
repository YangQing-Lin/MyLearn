###### c++

```c++
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

int n;
int res;

void dfs(int k) {
    if (k == n) {
        res ++;
    }
    
    if (k > n) {
        return;
    }
    
    dfs(k + 1);
    dfs(k + 2);
}

int main() {
    cin >> n;
    
    dfs(0);
    
    cout << res << endl;
    
    return 0;
}
```
