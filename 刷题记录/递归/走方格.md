###### c++

```c++
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

int n, m;
int res = 0;

void dfs(int x, int y) {
    if (x == n && y == m) {
        res ++;
    }
    
    if (y < m) {
        dfs(x, y + 1);
    }
    
    if (x < n) {
        dfs(x + 1, y);
    }
}

int main() {
    cin >> n >> m;
    
    dfs(0, 0);
    
    cout << res << endl;
    
    return 0;
}
```
