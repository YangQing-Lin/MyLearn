#### <a href="https://leetcode.cn/problems/find-the-distance-value-between-two-arrays/">两个数组间的距离值</a>

-------------

```java
class Solution {
    public int findTheDistanceValue(int[] arr1, int[] arr2, int d) {
        Arrays.sort(arr2);
        int res = 0;
        for (int t : arr1) {
            int p = help(arr2, t);
            boolean flag = true;
            if (p < arr2.length) {
                flag &= arr2[p] - t > d;
            }
            if (p - 1 >= 0) {
                flag &= t - arr2[p - 1] > d;
            }
            res += flag ? 1 : 0;
        }
        return res;
    }

    public int help(int[] arr, int t) {
        int l = 0, r = arr.length - 1;
        if (t > arr[r]) {
            return r + 1;
        }
        while (l < r) {
            int mid = l + (r - l) / 2;
            if (arr[mid] >= t) r = mid;
            else l = mid + 1;
        }
        return l;
    }
}
```
