#### <a href="https://www.acwing.com/problem/content/3514/">倒水问题</a>

--------------------

###### java

```java
import java.util.*;

public class Main {
    static Set<Integer> set = new HashSet<>();
    static Set<String> visited = new HashSet<>();
    static int A, B, C;
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        while (sc.hasNext()) {
            set.clear();
            visited.clear();
            A = sc.nextInt();
            B = sc.nextInt();
            C = sc.nextInt();
            dfs(0, 0, C);
            System.out.println(set.size());
        }
    }

    static void dfs(int a, int b, int c) {
        String cur = a+"   "+b+"   "+c;
        if (visited.contains(cur)) return;
        visited.add(cur);
        set.add(c);
        if (a > 0) {
            int min = Math.min(a, B - b);
            dfs(a - min, b + min, c);
            min = Math.min(a, C - c);
            dfs(a - min, b, c + min);
        }
        if (b > 0) {
            int min = Math.min(b, A - a);
            dfs(a + min, b - min, c);
            min = Math.min(b, C - c);
            dfs(a, b - min, c + min);
        }
        if (c > 0) {
            int min = Math.min(c, A - a);
            dfs(a + min, b, c - min);
            min = Math.min(c, B - b);
            dfs(a, b + min, c - min);
        }
    }
}
```
