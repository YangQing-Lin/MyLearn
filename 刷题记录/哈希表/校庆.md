###### c++

```c++
#include <iostream>
#include <cstring>
#include <algorithm>
#include <unordered_set>

using namespace std;

int main() {
    int n;
    cin >> n;
    
    unordered_set<string> set;
    while (n --) {
        string str_name;
        cin >> str_name;
        set.insert(str_name);
    }
    
    int m;
    cin >> m;
    
    string a; // 存交集（既是校友也是来宾）中年龄最大的
    string b;// 存来宾里年龄最大的
    int res = 0;
    while (m --) {
        string str_name;
        cin >> str_name;
        if (set.count(str_name)) {
            res ++;
            if (a.empty() || str_name.substr(6, 8) < a.substr(6, 8)) {
                a = str_name;
            }
        }
        if (b.empty() || str_name.substr(6, 8) < b.substr(6, 8)) {
            b = str_name;
        }
    }
    
    cout << res << endl;
    if (res) {
        cout << a << endl;
    }
    else {
        cout << b << endl;
    }
    
    return 0;
}
```
