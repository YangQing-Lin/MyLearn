#### 题目1：<a href="https://www.acwing.com/problem/content/848/">树的重心</a>

------------

##### 思路：

枚举每个点删除之后，剩余连通块中，点数最大值，找出点数最大值中的最小值，则删除的此点为重心

使用树的深度优先遍历求删掉后连通块中的点数最大值，因为深度优先遍历可以算出来每个子树的大小，

每个子树的大小算出之后，就可以通过 n - size 算出父节点子树的大小![树的重心](C:\Users\冬黎\OneDrive\图片\算法基础课\算法基础课第三讲\树的重心.png)

###### c++

```c++
#include <cstring>
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 100010, M = N * 2;

int n;
int h[N], e[M], ne[M], idx;
bool st[N];

int ans = N;

void add(int a, int b) {
    e[idx] = b;
    ne[idx] = h[a];
    h[a] = idx ++;
}

// 以 u 为根的子树中点的数量
int dfs(int u) {
    st[u] = true; // 标记一下，已经被搜过了，因为深度优先搜索和广度优先搜索都只会遍历一次
    
    int sum = 1, res = 0;
    for (int i = h[u]; i != -1; i = ne[i]) {
        int j = e[i];
        if (!st[j]) {
            int s = dfs(j);
            res = max(res,s);
            sum += s;
        }
    }
    
    res = max(res, n - sum);
    
    ans = min(ans, res);
    
    return sum;
}

int main() {
    cin >> n;
    
    memset(h, -1, sizeof(h));
    
    for (int i = 0; i < n - 1; i ++) {
        int a, b;
        cin >> a >> b;
        add(a, b), add(b, a);
    }
    
    dfs(1);
    
	cout << ans << endl;
    
    return 0;
}
```

```java
import java.util.LinkedList;
import java.util.List;
import java.util.Scanner;

public class Main {

    static int ans;

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = scanner.nextInt();
        ans = n + 1;
        Object[] tree = new Object[n + 1];
        boolean[] state = new boolean[n + 1];
        //构造树
        for (int i = 0; i < n - 1; i++) {
            int a = scanner.nextInt(), b = scanner.nextInt();
            if (tree[a] == null) {
                List<Integer> list = new LinkedList<>();
                list.add(b);
                tree[a] = list;
            } else {
                List<Integer> list = (List<Integer>) tree[a];
                list.add(b);
            }
            //边是无向的，所以需要添加两条边
            if (tree[b] == null) {
                List<Integer> list = new LinkedList<>();
                list.add(a);
                tree[b] = list;
            } else {
                List<Integer> list = (List<Integer>) tree[b];
                list.add(a);
            }

        }
        dfs(1, n, tree, state);
        System.out.println(ans);
        scanner.close();
    }

    // 返回以u为根的子树的大小
    private static int dfs(int u, int n, Object[] tree, boolean[] state) {
        state[u] = true;
        List<Integer> list = (List<Integer>) tree[u];
        if (list == null)
            return 1;
        int cont = 1;
        int res = 0;
        for (int i : list) {
            if (!state[i]) {
                int child = dfs(i, n, tree, state); 
                res = Math.max(res, child);  
                cont += child;
            }
        }
        res = Math.max(res, n - cont); 
        ans = Math.min(ans, res);
        return cont;
    }
}
```
