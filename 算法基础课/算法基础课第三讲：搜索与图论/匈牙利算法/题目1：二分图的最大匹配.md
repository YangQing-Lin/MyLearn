#### 题目1：<a href="https://www.acwing.com/problem/content/863/">二分图的最大匹配</a>

----------

##### 题解：

###### c++

```c++
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 510, M = 100010;

int n1, n2, m;
int h[N], e[M], ne[M], idx;
int match[N]; // 表示右边点匹配左边的点
bool st[N]; // 标记左边点每个考虑过的点

void add(int a, int b) {
    e[idx] = b;
    ne[idx] = h[a];
    h[a] = idx ++;
}

bool find(int k) {
    for (int i = h[k]; i != -1; i = ne[i]) { 
        int j = e[i];
        if (!st[j]) {
            st[j] = true;
            if (match[j] == 0 || find(match[j])) { // 如果当前点没有匹配过或者已经匹配的点可以找到其他的点匹配
                match[j] = k;
                return true;
            }
        }
    }
    
    return false;
}

int main() {
    cin >> n1 >> n2 >> m;
    
    memset(h, -1, sizeof(h));
    
    while (m --) {
        int a, b;
        cin >> a >> b;
        
        add(a, b);
    }
    
    int res = 0; // 记录最大匹配数
    for (int i = 1; i <= n1; i ++) {
        memset(st, false, sizeof(st)); // 保证每个女生只考虑一次
        if (find(i)) {
            res ++;
        }
    }
    
    printf("%d\n", res);
    
    return 0;
}
```

###### java

```java
import java.util.Scanner;
import java.util.Arrays;

class Main {
    static int n1, n2, m, idx;
    static int N = 510;
    static int M = 100010;
    static int[] h = new int[N];
    static int[] e = new int[M];
    static int[] ne = new int[M];
    static int[] match = new int[N];
    static boolean[] st = new boolean[N];
    
    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        n1 = scan.nextInt();
        n2 = scan.nextInt();
        m = scan.nextInt();
        
        Arrays.fill(h, -1);
        
        while (m -- > 0) {
            int a, b;
            a = scan.nextInt();
            b = scan.nextInt();
            
            add(a, b);
        }
        
        int res = 0;
        for (int i = 1; i <= n1; i ++) {
            Arrays.fill(st, false);
            if (find(i)) {
                res ++;
            } 
        }
        
        System.out.println(res);
    }
    
    public static void add(int a, int b) {
        e[idx] = b;
        ne[idx] = h[a];
        h[a] = idx ++;
    }
    
    public static boolean find(int k) {
        for (int i = h[k]; i != -1; i = ne[i]) {
            int j = e[i];
            if (!st[j]) {
                st[j] = true;
                if (match[j] == 0 || find(match[j])) {
                    match[j] = k;
                    return true;
                }
            }
        }
        
        return false;
    }
}
```
