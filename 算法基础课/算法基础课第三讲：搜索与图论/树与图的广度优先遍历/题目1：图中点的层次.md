###### c++

```c++
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 100010;

int n, m;
int h[N], e[N], ne[N], idx;
int d[N], q[N];

void add(int a, int b) {
    e[idx] = b;
    ne[idx] = h[a];
    h[a] = idx;
    idx ++;
}

int bfs() {
    int hh = 0, tt = 0;
    q[0] = 1;
    
    memset(d, -1, sizeof(d));
    
    d[1] = 0;
    
    while (hh <= tt) {
        int t = q[hh ++];
        
        for (int i = h[t]; i != -1; i = ne[i]) {
            int j = e[i];
            if (d[j] == -1) {
                d[j] = d[t] + 1;
                q[++ tt] = j;
            }
        }
    }
    
    return d[n];
}

int main() {
    cin >> n >> m;
    
    memset(h, -1, sizeof(h));
    
    for (int i = 0; i < m; i ++) {
        int a, b;
        cin >> a >> b;
        
        add(a, b);
    }
    
    cout << bfs() << endl;
    
    return 0;
}
```

###### java

```java
import java.util.Scanner;
import java.util.Queue;
import java.util.LinkedList;

class Main {
    static int N = 100010;
    static int idx, n;
    static int[] h = new int[N];
    static int[] e = new int[N];
    static int[] ne = new int[N];
    static int[] d = new int[N];
    static Queue<Integer> q = new LinkedList<>();
    
    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        n = scan.nextInt();
        int m = scan.nextInt();
        
        for (int i = 0; i < N; i ++) {
            d[i] = -1;
            h[i] = -1;
        }
        
        for (int i = 0; i < m; i ++) {
            int a, b;
            a = scan.nextInt();
            b = scan.nextInt();
            
            add(a, b);
        }
        
        System.out.print(bfs());
    }
    
    public static void add(int a, int b) {
        e[idx] = b;
        ne[idx] = h[a];
        h[a] = idx;
        idx ++;
    }
    
    public static int bfs() {
        q.offer(1);
        d[1] = 0;
        
        while (q.size() != 0) {
            int t = q.poll();
            
            for (int i = h[t]; i != -1; i = ne[i]) {
                int j = e[i];
                if (d[j] == -1) {
                    d[j] = d[t] + 1;
                    q.offer(j);
                }
            }
        }
        
        return d[n];
    }
}
```
