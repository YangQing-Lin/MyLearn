#### 题目1：<a href="https://www.acwing.com/problem/content/850/">有向图的拓扑排序</a>

--------------

##### 题解：

###### c++

```c++
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 100010;

int n, m;
int h[N], e[N], ne[N], idx;
int q[N], d[N];

void add(int a, int b) {
    e[idx] = b;
    ne[idx] = h[a];
    h[a] = idx;
    idx ++;
}

bool topsort() {
    int hh = 0, tt = -1;
    
    for (int i = 1; i <= n; i ++) {
        if (d[i] == 0) {
            q[++ tt] = i;
        }
    }
    
    while (hh <= tt) {
        int t = q[hh ++];
        
        for (int i = h[t]; i != -1; i = ne[i]) {
            int j = e[i];
            d[j] --;
            if (d[j] == 0) {
                q[++ tt] = j;
            }
        }
    }
    
    return tt == n - 1;
}

int main() {
    cin >> n >> m;
    
    memset(h, -1, sizeof(h));
    
    for (int i = 0; i < m; i ++) {
        int a, b;
        cin >> a >> b;
        add(a, b);
        d[b] ++;
    }
    
    if (topsort()) {
        for (int i = 0; i < n; i ++) {
            printf("%d ", q[i]);
        }
    }
    else {
        puts("-1");
    }
    
    return 0;
}
```

###### java

import java.util.Scanner;
import java.util.Queue;
import java.util.LinkedList;

class Main {
    static int n,m;
    static int idx;

    static int[] h = new int[N];
    static int[] e = new int[N];
    static int[] ne = new int[N];
    static int[] d = new int[N];
    static int[] q = new int[N];
    
    public static void main(String[] args) {
        Scanner scan = new Scanner(System.in);
        n = scan.nextInt();
        m = scan.nextInt();
        
        for (int i = 0; i < N; i ++) {
            h[i] = -1;
        }
        
        for (int i = 0; i < m; i ++) {
            int a = scan.nextInt();
            int b = scan.nextInt();
            add(a, b);
            d[b] ++;
        }
        
        if (topsort()) {
            for (int i = 0; i < n; i ++) {
                System.out.print(q[i] + " ");
            }
        }
        else {
            System.out.print("-1");
        }
    }
    
    public static void add(int a, int b) {
        e[idx] = b;
        ne[idx] = h[a];
        h[a] = idx;
        idx ++;
    }
    
    public static boolean topsort() {
        int hh = 0, tt = -1;
        for (int i = 1; i <= n; i ++) {
            if (d[i] == 0) {
                q[++ tt] = i;
            }
        }
        
        while (hh <= tt) {
            int t = q[hh ++];

                int j = e[i];
                d[j] --;

                }
            }
        }

    }
}
