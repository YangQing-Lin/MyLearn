#### 题目1：<a href="https://www.acwing.com/problem/content/846/">走迷宫</a>

--------------

##### 题解：

###### c++

```c++
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

typedef pair<int, int> PII;

const int N = 110;

int n, m;
int g[N][N]; // 存地图
int d[N][N]; // 存每个点到起点的距离
PII q[N * N]; // 存最短路径

int bfs(){
    int hh = 0, tt = 0; // 初始化队列
    q[0] = {0, 0};
    
    memset(d, -1, sizeof(d)); // 初始化距离
    
    d[0][0] = 0; // 表示起点是（0，0）
    
    int dx[4] = {-1, 1, 0, 0}, dy[4] = {0, 0, -1, 1}; // 向量表示上下左右方向的点
    
    while (hh <= tt) {
        auto t = q[hh ++]; // 拿出队头
        
        for (int i = 0; i < 4; i ++) { // 遍历队头上下左右四个点
            int x = t.first + dx[i], y = t.second + dy[i]; // 表示每个点的横纵坐标
            if (x >= 0 && x < n && y >= 0 && y < m && d[x][y] == -1 && g[x][y] == 0) {
                d[x][y] = d[t.first][t.second] + 1; 
                q[++ tt] = {x, y};
            }
        }
    }
    
    return d[n - 1][m - 1];
}

int main(){
    cin >> n >> m;
    
    for (int i = 0; i < n; i ++) {
        for (int j = 0; j < m; j ++) {
            cin >> g[i][j];
        }
    }
    
    cout << bfs() << endl;
    
    return 0;
}
```

###### java

```java
import java.util.*;

public class Main {
    static int n, m;
    static int[][] g, d;
    static int[][] dirs = {{0, 1}, {0, -1}, {1, 0}, {-1, 0}}; // 定义队头的上下左右四个点
    static class Pair {
        int x, y;
        public Pair(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        n = sc.nextInt();
        m = sc.nextInt();
        g = new int[n][m];
        d = new int[n][m];

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                g[i][j] = sc.nextInt();
            }
        }

        System.out.println(bfs());
    }

    private static int bfs() {
        Queue<Pair> q = new LinkedList<>();

        q.offer(new Pair(0, 0));
        while (!q.isEmpty()) {
            Pair pair = q.poll();

            for (int[] dir : dirs) {
                int x = pair.x + dir[0];
                int y = pair.y + dir[1];
                if (x < 0 || y < 0 || x >= n || y >= m || g[x][y] == 1 || d[x][y] != 0) continue;
                d[x][y] = d[pair.x][pair.y] + 1;
                q.offer(new Pair(x, y));
            }
        }

        return d[n - 1][m - 1];
    }
}
```
