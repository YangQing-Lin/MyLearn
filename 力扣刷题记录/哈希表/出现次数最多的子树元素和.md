#### <a href="https://leetcode.cn/problems/most-frequent-subtree-sum/submissions/">出现次数最多的子树元素和</a>

------------

###### c++

```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
private:
    unordered_map<int, int>  map;
    int k = 0;

public:
    void dfs(TreeNode* root) { // 深搜，并把每个结点的值作为 key 该结点的子树元素和作为 value 加入哈希表
        if (root == nullptr) {
            return;
        }

        map.insert(pair<int, int>(k ++, TreeSum(root)));
        if (root -> left) {
            dfs(root -> left);
        }
        if (root -> right) {
            dfs(root -> right);
        }
    }

    int TreeSum(TreeNode* root) { // 结点的子树元素和
        if (root == nullptr) {
            return 0;
        }
        int res = 0;
        res += TreeSum(root -> left) + TreeSum(root -> right) + root -> val;

        return res;
    }

    vector<int> findFrequentTreeSum(TreeNode* root) {
        vector<int> ans;
        if (root == nullptr) {
            return ans;
        }

        dfs(root);

        for (auto p : map) {
            ans.push_back(p.second);
        }

        int tem = 0;
        for (int i = 0; i < ans.size(); i ++) { // 找到出现次数最多的子树元素和
            int b = count(ans.begin(), ans.end(), ans[i]);
            tem = max(tem, b);
        }

        vector<int> a;
        for (int i = 0; i < ans.size(); i ++) {
            if (count(ans.begin(), ans.end(), ans[i]) == tem) {
                a.push_back(ans[i]);
            }
        }
        
        sort(a.begin(), a.end());
        a.erase(unique(a.begin(), a.end()), a.end());

        return a;
    }
};
```
