#### <a href="https://leetcode.cn/problems/symmetric-tree/submissions/">对称二叉树</a>

--------------

###### c++

```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool check(TreeNode* q, TreeNode* p) {
        if (!q && !p) {
            return true;
        }
        if (!q || !p) {
            return false;
        }

        return q -> val == p -> val && check(q -> left, p -> right) && check(q -> right, p -> left);
    }

    bool isSymmetric(TreeNode* root) {
        return check(root, root);
    }
};
```

###### java

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean check(TreeNode q, TreeNode p) {
        if (q == null && p == null) {
            return true;
        }
        if (q == null || p == null) {
            return false;
        }

        return q.val == p.val && check(q.left, p.right) && check(q.right, p.left);
    }

    public boolean isSymmetric(TreeNode root) {
        return check(root, root);
    }
}
```
